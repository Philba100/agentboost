'use client';

import { useSearchParams } from 'next/navigation';
import { useEffect, useState, Suspense, useCallback } from 'react';
import { createClient, User } from '@supabase/supabase-js';
import Link from 'next/link';

// Enterprise Standard: Strict TypeScript Interfaces
interface Profile {
  id: string;
  subscription_tier: 'free' | 'pro' | 'enterprise';
  created_at?: string;
}

// Enterprise Standard: Singleton Supabase client initialization
const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!
);

// 🛑 Stripe Price ID: Configure via environment variable
const LEGAL_PRICE_ID = process.env.NEXT_PUBLIC_LEGAL_PRICE_ID || "";

function LegalRiskDashboard() {
  const searchParams = useSearchParams();
  const [loading, setLoading] = useState<boolean>(true);
  const [user, setUser] = useState<User | null>(null);
  const [profile, setProfile] = useState<Profile | null>(null);
  const [processing, setProcessing] = useState<boolean>(false);
  const [error, setError] = useState<string | null>(null);

  // Fallbacks and input sanitization
  const contractType = searchParams.get('contract_type') || 'Master Services Agreement';
  const focusClause = searchParams.get('focus_clause') || 'Indemnification';

  const fetchSessionAndProfile = useCallback(async () => {
    try {
      const { data: { session }, error: sessionError } = await supabase.auth.getSession();
      
      if (sessionError) throw sessionError;
      
      const currentUser = session?.user || null;
      setUser(currentUser);

      if (currentUser) {
        const { data: profileData, error: profileError } = await supabase
          .from('profiles')
          .select('*')
          .eq('id', currentUser.id)
          .single();
          
        if (profileError) throw profileError;
        setProfile(profileData);
      }
    } catch (err: any) {
      console.error('Auth/Profile Fetch Error:', err.message);
      setError('Failed to load user data. Please refresh.');
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => {
    fetchSessionAndProfile();
  }, [fetchSessionAndProfile]);

  const handleUpgrade = async () => {
    setProcessing(true);
    setError(null);
    try {
      const res = await fetch('/api/checkout', { 
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ 
          email: user?.email, 
          userId: user?.id,
          priceId: LEGAL_PRICE_ID
        })
      });
      
      if (!res.ok) throw new Error('Failed to initiate secure checkout.');
      
      const data = await res.json();
      if (data.url) {
        window.location.href = data.url;
      } else {
        throw new Error('Invalid checkout URL received.');
      }
    } catch (err: any) {
      console.error('Checkout Error:', err);
      setError(err.message || 'Payment service unavailable.');
      setProcessing(false);
    }
  };

  if (loading) {
    return (
      <div className="min-h-screen bg-[#0f172a] text-red-400 flex items-center justify-center font-bold font-mono text-sm animate-pulse">
        RUNNING NLP CONTRACT ANALYSIS FOR {contractType.toUpperCase()}...
      </div>
    );
  }

  if (!user) {
    return (
      <main className="min-h-screen bg-[#0f172a] text-white flex flex-col items-center justify-center p-8 text-center">
        <section className="bg-[#1e293b] p-10 rounded-2xl border border-red-500/20 max-w-md w-full shadow-2xl relative overflow-hidden">
          <div className="absolute top-0 left-1/2 -translate-x-1/2 w-full h-1 bg-gradient-to-r from-transparent via-red-500 to-transparent opacity-30"></div>
          <div className="w-12 h-12 bg-red-500/10 rounded-lg flex items-center justify-center mx-auto mb-6 border border-red-500/20">
            <svg className="w-6 h-6 text-red-400" fill="none" viewBox="0 0 24 24" stroke="currentColor" strokeWidth={1.5}><path strokeLinecap="round" strokeLinejoin="round" d="M12 9v3.75m-9.303 3.376c-.866 1.5.217 3.374 1.948 3.374h14.71c1.73 0 2.813-1.874 1.948-3.374L13.949 3.378c-.866-1.5-3.032-1.5-3.898 0L2.697 16.126zM12 15.75h.007v.008H12v-.008z" /></svg>
          </div>
          <h2 className="text-2xl font-bold mb-2 tracking-tight">Legal Counsel Pro</h2>
          <p className="text-slate-400 mb-8 text-sm">Authentication required to view compliance risks and hidden liabilities.</p>
          <Link href="/login?next=/legal-risk" className="block w-full bg-red-600 hover:bg-red-500 transition-colors text-white font-bold py-3.5 rounded-md">
            Authorize Session
          </Link>
        </section>
      </main>
    );
  }

  const isPro = profile?.subscription_tier === 'pro' || profile?.subscription_tier === 'enterprise';

  return (
    <main className="min-h-screen bg-[#0f172a] text-slate-200 p-6 font-sans">
      <header className="flex justify-between items-center border-b border-slate-800 pb-4 mb-6">
        <div>
          <h1 className="text-xl font-bold">Risk Assessment</h1>
          <p className="text-red-400 text-[10px] font-bold tracking-widest uppercase">Doc: {contractType}</p>
        </div>
        <span className={`px-2 py-1 rounded text-[10px] font-bold border ${isPro ? 'bg-red-500/10 text-red-400 border-red-500/30' : 'bg-slate-800 text-slate-500 border-slate-700'}`}>
          {isPro ? 'PRO ACTIVE' : 'STARTER'}
        </span>
      </header>

      {error && (
        <div className="mb-6 p-4 bg-red-900/50 border border-red-500/50 text-red-200 rounded-xl text-sm">
          {error}
        </div>
      )}

      {!isPro ? (
        <div className="space-y-6">
          <div className="bg-slate-900/50 p-5 rounded-xl border border-slate-800 select-none">
             <div className="flex justify-between items-center mb-4">
                <div className="h-4 w-32 bg-slate-800 rounded animate-pulse"></div>
                <div className="h-4 w-16 bg-slate-800 rounded animate-pulse"></div>
             </div>
             <div className="h-6 w-full bg-slate-800 rounded mb-2 blur-md"></div>
             <div className="h-4 w-3/4 bg-slate-800 rounded blur-md"></div>
          </div>

          <section className="bg-gradient-to-br from-[#1e293b] to-red-900/20 p-8 text-center rounded-2xl border border-red-500/30 shadow-2xl">
            <h3 className="font-black text-2xl mb-2">Reveal Hidden Liabilities</h3>
            <p className="text-slate-400 text-sm mb-8">
              Unlock deep-tier legal analysis, liability traps, and rewritten safe clauses for <span className="text-white font-semibold">{focusClause}</span>.
            </p>
            <button 
              onClick={handleUpgrade} 
              disabled={processing} 
              className="w-full bg-red-600 hover:bg-red-500 disabled:bg-red-800 transition-all text-white font-bold py-4 rounded-xl shadow-lg shadow-red-500/20"
            >
              {processing ? 'Authorizing...' : 'Upgrade to Counsel Pro — $49.99/mo'}
            </button>
          </section>
        </div>
      ) : (
        <div className="space-y-4">
          <article className="bg-[#1e293b] p-5 rounded-2xl border border-red-500/30 shadow-lg">
            <div className="flex justify-between items-start mb-4">
               <div>
                  <h3 className="text-lg font-bold text-red-400">Critical Risk Detected</h3>
                  <p className="text-slate-400 text-xs font-medium uppercase tracking-tighter">Clause: {focusClause}</p>
               </div>
               <span className="bg-red-500/10 text-red-500 text-[10px] font-bold px-2 py-1 rounded border border-red-500/20">HIGH LIABILITY</span>
            </div>
            <div className="space-y-3 font-mono text-sm">
               <div className="border-b border-slate-800 pb-3">
                  <span className="block text-slate-500 mb-1 text-xs uppercase">Original Text (Snippet)</span>
                  <span className="text-white bg-red-900/30 px-2 py-1 rounded">"...shall indemnify and hold harmless against ALL claims..."</span>
               </div>
               <div>
                  <span className="block text-slate-500 mb-1 text-xs uppercase">Suggested Revision</span>
                  <span className="text-emerald-400">"...shall indemnify against direct, proven claims arising solely from gross negligence..."</span>
               </div>
            </div>
          </article>
          
          <button className="w-full bg-slate-100 hover:bg-white transition-colors text-slate-900 font-bold py-4 rounded-xl text-sm shadow-md">
            Export Redlined PDF
          </button>
        </div>
      )}
    </main>
  );
}

export default function Page() {
  return (
    <Suspense fallback={<div className="min-h-screen bg-[#0f172a]" />}>
      <LegalRiskDashboard />
    </Suspense>
  );
}
